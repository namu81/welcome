async function saveLog() { const company = document.getElementById('company').value; const userName = document.getElementById('userName').value; const note = document.getElementById('note').value; const type = document.querySelector('input[name="type"]:checked').value; if (!company || !userName) { alert('기업명과 사용자를 입력해주세요.'); return; } try { const response = await fetch(`${SUPABASE_URL}/rest/v1/messages`, { method: 'POST', headers: { 'apikey': SUPABASE_KEY, 'Authorization': `Bearer ${SUPABASE_KEY}`, 'Content-Type': 'application/json', 'Prefer': 'return=minimal' }, body: JSON.stringify({ company: company, user_name: userName, note: note, type: type }) }); if (response.ok) { alert('저장 성공!'); document.getElementById('company').value = ''; document.getElementById('userName').value = ''; document.getElementById('note').value = ''; displayLogs(); } else { const errorData = await response.json(); console.error('저장 실패:', errorData); alert('저장 실패: ' + errorData.message); } } catch (err) { console.error('네트워크 에러:', err); alert('네트워크 에러가 발생했습니다.'); } }
